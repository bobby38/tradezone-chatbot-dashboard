Installation Code

  data-api-url="https://your-dashboard.com"
  data-position="bottom-right"

Manual Initialization

<script src="https://your-domain.com/widget/chat-widget-enhanced.js"></script>
<script>
  TradeZoneChatEnhanced.init({
    apiUrl: 'https://your-dashboard.com',
    position: 'bottom-right',
    primaryColor: '#2563eb',
    videoUrl: 'https://your-domain.com/avatar-video.mp4',
    greeting: 'Hi! I can help via text or voice. What do you prefer?',
    botName: 'Izacc',
    enableVoice: true,  // Enable voice chat
    enableVideo: true   // Enable video hero
  });
</script>

Configuration Options

{
  apiUrl: 'https://your-dashboard.com',  // Required
  position: 'bottom-right',              // or 'bottom-left'
  primaryColor: '#2563eb',               // Brand color
  videoUrl: 'https://...',               // Hero video URL
  greeting: 'Hi! How can I help?',       // Initial message
  botName: 'Amara',                      // Bot display name
  placeholder: 'Type your message...',   // Input placeholder
  enableVoice: true,                     // Enable voice chat
  enableVideo: true                      // Enable video hero
}
